# iOS PWA Core Web IM Phase 1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the first independently runnable `web_im/` Vue 3 PWA shell with fake IM data, mobile-first navigation, virtual chat list foundation, IndexedDB persistence boundary, service worker shell, and deploy/test hooks.

**Architecture:** Create a new `web_im/` app instead of mixing customer IM into Flutter Web or the admin console. Phase 1 uses fake data and local browser APIs only so mobile layout, long-list rendering, IndexedDB shape, service worker scope, and deployment packaging can be validated before real backend integration.

**Tech Stack:** Vue 3, TypeScript, Vite, Pinia, Vue Router, Vitest, Playwright, native IndexedDB, native Service Worker APIs, pnpm.

---

## Scope

This plan implements Phase 1 from the approved design:

- `web_im/` independent PWA project.
- Fake-data login and IM workspace.
- Mobile-first routes: login, conversations, contacts, chat, me.
- DOM virtual message list foundation with anchor-preserving calculations.
- IndexedDB adapter and repository boundary for conversations, messages, and drafts.
- Service worker shell with offline fallback and notification-click routing helper.
- Build and deploy script hooks for `/im`.
- Unit/component/e2e smoke tests.

This plan does not connect real backend APIs, real WebSocket, real file upload, or real Web Push subscription delivery. Those belong to later Phase 2-4 plans.

- [ ] **Step 4: Create auth store**

Create `web_im/src/stores/authStore.ts`:

```ts
import { defineStore } from 'pinia';
import { computed, ref } from 'vue';
import { fakeCurrentUser } from '@/mocks/fakeImData';

const TOKEN_KEY = 'wk_web_im_fake_token';

export const useAuthStore = defineStore('auth', () => {
  const token = ref(window.localStorage.getItem(TOKEN_KEY) || '');
  const user = ref(token.value ? fakeCurrentUser : null);
  const isLoggedIn = computed(() => Boolean(token.value && user.value));

  function login(phone: string, password: string) {
    if (!/^1\d{10}$/.test(phone.trim())) {
      throw new Error('请输入 11 位手机号');
    }
    if (password.trim().length < 4) {
      throw new Error('请输入密码');
    }
    token.value = `fake-token-${Date.now()}`;
    window.localStorage.setItem(TOKEN_KEY, token.value);
    user.value = { ...fakeCurrentUser, phone };
  }

  function logout() {
    token.value = '';
    user.value = null;
    window.localStorage.removeItem(TOKEN_KEY);
  }

  return { token, user, isLoggedIn, login, logout };
});
```

Create `web_im/src/app/router.ts`:

```ts
import { createRouter, createWebHistory, type RouteRecordRaw } from 'vue-router';
import { useAuthStore } from '@/stores/authStore';
import LoginPage from '@/features/login/LoginPage.vue';
import MobileShell from '@/features/shell/MobileShell.vue';
import ConversationListPage from '@/features/conversations/ConversationListPage.vue';
import ContactsPage from '@/features/contacts/ContactsPage.vue';
import MePage from '@/features/me/MePage.vue';
import ChatPage from '@/features/chat/ChatPage.vue';

export const routes: RouteRecordRaw[] = [
  { path: '/login', component: LoginPage },
  {
    path: '/',
    component: MobileShell,
    redirect: '/conversations',
    children: [
      { path: '/conversations', component: ConversationListPage },
      { path: '/contacts', component: ContactsPage },
      { path: '/me', component: MePage },
      { path: '/chat/:channelType/:channelId', component: ChatPage },
    ],
  },
];

export const router = createRouter({
  history: createWebHistory(import.meta.env.BASE_URL),
  routes,
});

router.beforeEach((to) => {
  const auth = useAuthStore();
  if (!auth.isLoggedIn && to.path !== '/login') {
    return '/login';
  }
  if (auth.isLoggedIn && to.path === '/login') {
    return '/conversations';
  }
  return true;
});
```

Create `web_im/src/features/chat/virtualList.ts`:

```ts
export interface VirtualWindowInput {
  itemCount: number;
  rowHeight: number;
  scrollTop: number;
  viewportHeight: number;
  overscan: number;
}

export interface VirtualWindow {
  start: number;
  end: number;
  beforeHeight: number;
  afterHeight: number;
}

export const computeVirtualWindow = (input: VirtualWindowInput): VirtualWindow => {
  const itemCount = Math.max(0, input.itemCount);
  if (itemCount === 0) {
    return { start: 0, end: 0, beforeHeight: 0, afterHeight: 0 };
  }

  const rowHeight = Math.max(1, input.rowHeight);
  const viewportHeight = Math.max(1, input.viewportHeight);
  const overscan = Math.max(0, input.overscan);
  const firstVisible = Math.floor(Math.max(0, input.scrollTop) / rowHeight);
  const visibleCount = Math.ceil(viewportHeight / rowHeight);
  const start = Math.max(0, firstVisible - overscan);
  const end = Math.min(itemCount, firstVisible + visibleCount + overscan);
  return {
    start,
    end,
    beforeHeight: start * rowHeight,
    afterHeight: Math.max(0, itemCount - end) * rowHeight,
  };
};

export const preservePrependAnchor = ({
  previousScrollTop,
  insertedCount,
  rowHeight,
}: {
  previousScrollTop: number;
  insertedCount: number;
  rowHeight: number;
}) => Math.max(0, previousScrollTop) + Math.max(0, insertedCount) * Math.max(1, rowHeight);
```

```ts
export const useChatStore = defineStore('chat', () => {
  const messagesByChannel = ref<Record<string, ChatMessage[]>>({});
  const activeChannelKey = ref('');

  const activeMessages = computed(() => messagesByChannel.value[activeChannelKey.value] || []);
  const activeConversation = computed(() =>

  );

  function openChannel(channelType: ChannelType, channelId: string) {
    const key = `${channelType}_${channelId}`;
    activeChannelKey.value = key;
    if (!messagesByChannel.value[key]) {
      messagesByChannel.value[key] = buildFakeMessages(channelId);
    }
  }

  function sendText(text: string) {
    const normalized = text.trim();
    if (!normalized || !activeChannelKey.value) {
      return;
    }
    const [channelType, channelId] = activeChannelKey.value.split('_');
    const next: ChatMessage = {
      id: `local-${Date.now()}`,
      clientMsgNo: `local-${Date.now()}`,
      channelId,
      channelType: Number(channelType) as ChannelType,
      fromUid: fakeCurrentUser.uid,
      fromName: fakeCurrentUser.name,
      direction: 'outgoing',
      kind: 'text',
      text: normalized,
      timestamp: Date.now(),
      status: 'sent',
    };
    messagesByChannel.value[activeChannelKey.value] = [...activeMessages.value, next];
  }

  function prependOlderMessages() {
    if (!activeChannelKey.value) {
      return 0;
    }
    const current = activeMessages.value;
    const older = Array.from({ length: 20 }, (_, index): ChatMessage => ({
      ...current[0],
      id: `older-${Date.now()}-${index}`,
      clientMsgNo: `older-${Date.now()}-${index}`,
      text: `更早的历史消息 ${index + 1}`,
      timestamp: (current[0]?.timestamp || Date.now()) - (20 - index) * 60 * 1000,
      direction: index % 3 === 0 ? 'outgoing' : 'incoming',
      status: 'sent',
    }));
    messagesByChannel.value[activeChannelKey.value] = [...older, ...current];
    return older.length;
  }

  return { activeMessages, activeConversation, openChannel, sendText, prependOlderMessages };
```

```ts
export interface NotificationPayload {
  url?: unknown;
  channel_id?: unknown;
  channelId?: unknown;
  channel_type?: unknown;
  channelType?: unknown;
  payload?: Record<string, unknown>;
}

const fallback = '/im/conversations';

const readString = (...values: unknown[]) => {
  for (const value of values) {
    if (typeof value === 'string' && value.trim()) {
      return value.trim();
    }
  }
  return '';
};

const readChannelType = (...values: unknown[]) => {
  for (const value of values) {
    const text = typeof value === 'number' ? String(Math.trunc(value)) : typeof value === 'string' ? value.trim() : '';
    if (/^\d+$/.test(text)) {
      return text;
    }
  }
  return '';
};

export const resolveNotificationTarget = (data: NotificationPayload) => {
  const rawUrl = typeof data.url === 'string' ? data.url.trim() : '';
  if (rawUrl.startsWith('/im/')) {
    return rawUrl;
  }

  const payload = data.payload || {};
  const channelId = readString(data.channel_id, data.channelId, payload.channel_id, payload.channelId);
  const channelType = readChannelType(data.channel_type, data.channelType, payload.channel_type, payload.channelType);
  if (channelId && channelType) {
    return `/im/chat/${encodeURIComponent(channelType)}/${encodeURIComponent(channelId)}`;
  }
  return fallback;
};
```

```ts
export interface PwaLifecycleHandlers {
  onVisible?: () => void;
  onHidden?: () => void;
  onOnline?: () => void;
  onOffline?: () => void;
}

export const installPwaLifecycle = (handlers: PwaLifecycleHandlers) => {
  const handleVisibility = () => {
    if (document.visibilityState === 'visible') {
      handlers.onVisible?.();
    } else {
      handlers.onHidden?.();
    }
  };
  document.addEventListener('visibilitychange', handleVisibility);
  window.addEventListener('pageshow', handlers.onVisible || (() => undefined));
  window.addEventListener('pagehide', handlers.onHidden || (() => undefined));
  window.addEventListener('online', handlers.onOnline || (() => undefined));
  window.addEventListener('offline', handlers.onOffline || (() => undefined));

  return () => {
    document.removeEventListener('visibilitychange', handleVisibility);
  };
};
```

```powershell
[CmdletBinding()]
param(
    [string]$WebImDir = 'web_im'
)

Set-StrictMode -Version Latest
$ErrorActionPreference = 'Stop'

$root = (Resolve-Path -LiteralPath (Join-Path $PSScriptRoot '..\..')).Path
$resolvedWebImDir = (Resolve-Path -LiteralPath (Join-Path $root $WebImDir)).Path

Write-Host "[STEP] Installing Web IM dependencies"
pnpm --dir $resolvedWebImDir install --frozen-lockfile
if ($LASTEXITCODE -ne 0) {
    throw "pnpm install failed for Web IM."
}

Write-Host "[STEP] Building Web IM release"
pnpm --dir $resolvedWebImDir build
if ($LASTEXITCODE -ne 0) {
    throw "pnpm build failed for Web IM."
}

$dist = Join-Path $resolvedWebImDir 'dist'
foreach ($required in @('index.html', 'manifest.webmanifest', 'sw.js', 'offline.html')) {
    $path = Join-Path $dist $required
    if (-not (Test-Path -LiteralPath $path -PathType Leaf)) {
        throw "Missing required Web IM release artifact: $path"
    }
}

Write-Host "WEB_IM_RELEASE_DIR=$dist"
```
